# Gateway Support Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add first-class LLM gateway support (LiteLLM and others) to proxied-claude profiles, symmetric with the existing proxy subsystem.

**Architecture:** New `gateways/<n>.conf` resource type links to profiles via `PROFILE_GATEWAY`. On launch, `proxied-claude` reads the linked gateway, fetches the token from Keychain, and exports `ANTHROPIC_BASE_URL` + `ANTHROPIC_AUTH_TOKEN`. Management commands in `claude-proxy` follow the exact pattern of existing proxy commands.

**Tech Stack:** bash, bats-core (tests), macOS Keychain (`security`)

**Run tests:** `bats proxied-claude.bats`
**Run filtered:** `bats proxied-claude.bats --filter gateway`

- [ ] **Step 3: Implement `cmd_gateway` in `claude-proxy` — `create` subcommand**

Add `cmd_gateway` function after `cmd_proxy`. The `create` subcommand (note: prompts for token interactively, consistent with `proxy create`):

```bash
# ── gateway subcommands ──────────────────────────────────────────────────────

cmd_gateway() {
  local sub="${1:-}"; shift || true

  case "$sub" in

    create)
      require_interactive
      local name="${1:-}" url="${2:-}"
      [[ -n "$name" && -n "$url" ]] || \
        die "Usage: claude-proxy gateway create <n> <url>"
      validate_name "$name" "gateway name"
      [[ "$url" =~ ^https?:// ]] || \
        die "Invalid URL '$url'. Must start with http:// or https://"
      ensure_dirs
      local conf="$GATEWAYS_DIR/${name}.conf"
      local svc; svc="$(gateway_keychain_service "$name")"
      [[ ! -f "$conf" ]] || die "Gateway '$name' already exists. View: claude-proxy gateway show $name"
      echo "Creating gateway '$name' ($url)"
      echo "Enter gateway token (stored in macOS Keychain):"
      security add-generic-password -a "$name" -s "$svc" -U -w
      lock_acquire
      if [[ -f "$conf" ]]; then
        security delete-generic-password -a "$name" -s "$svc" 2>/dev/null || true
        lock_release
        die "Gateway '$name' already exists (created concurrently). View: claude-proxy gateway show $name"
      fi
      write_gateway_conf "$conf" "$url" "$svc"
      ok "Gateway '$name' created"
      info "Keychain service : $svc"
      info "Link to a profile: claude-proxy profile set-gateway <profile> $name"
      lock_release
      ;;

    *)
      cat <<'EOF'
Usage:
  claude-proxy gateway create <n> <url>
  claude-proxy gateway list
  claude-proxy gateway show  <n>
  claude-proxy gateway delete <n>
  claude-proxy gateway rename  <old> <new>
  claude-proxy gateway set-token <n>
EOF
      exit 1
      ;;
  esac
}
```

      if security delete-generic-password -a "$name" -s "$gsvc" >/dev/null 2>&1; then
        info "Keychain entry removed"
      else
        info "Keychain entry not found (skipped)"
      fi
      rm "$GATEWAYS_DIR/${name}.conf"
      ok "Gateway '$name' deleted"
      lock_release

```bash
    rename)
      lock_acquire
      local old="${1:-}" new="${2:-}"
      [[ -n "$old" && -n "$new" ]] || die "Usage: claude-proxy gateway rename <old> <new>"
      validate_name "$old" "gateway name"
      validate_name "$new" "gateway name"
      require_gateway "$old"
      [[ ! -f "$GATEWAYS_DIR/${new}.conf" ]] || die "Gateway '$new' already exists."

      local gurl; gurl="$(read_conf "$GATEWAYS_DIR/${old}.conf" GATEWAY_URL)"
      local old_svc; old_svc="$(read_conf "$GATEWAYS_DIR/${old}.conf" GATEWAY_KEYCHAIN_SERVICE)"
      old_svc="${old_svc:-$(gateway_keychain_service "$old")}"
      local new_svc; new_svc="$(gateway_keychain_service "$new")"

      # Migrate Keychain entry
      local token
      token="$(security find-generic-password -a "$old" -s "$old_svc" -w 2>/dev/null || true)"
      if [[ -n "$token" ]]; then
        security add-generic-password -a "$new" -s "$new_svc" -w "$token" -U >/dev/null 2>&1 || true
        security delete-generic-password -a "$old" -s "$old_svc" >/dev/null 2>&1 || true
        info "Keychain entry renamed: $old_svc → $new_svc"
      else
        warn "Could not migrate Keychain entry — run: claude-proxy gateway set-token $new"
      fi

      write_gateway_conf "$GATEWAYS_DIR/${new}.conf" "$gurl" "$new_svc"
      rm "$GATEWAYS_DIR/${old}.conf"

      # Update linked profiles
      shopt -s nullglob
      for pf in "$PROFILES_DIR"/*.conf; do
        local linked; linked="$(read_conf "$pf" PROFILE_GATEWAY)"
        if [[ "$linked" == "$old" ]]; then
          local pdir;  pdir="$(read_conf  "$pf" PROFILE_CLAUDE_DIR)"
          local proxy; proxy="$(read_conf "$pf" PROFILE_PROXY)"
          write_profile_conf "$pf" "$pdir" "$proxy" "$new"
          info "Updated profile '$(basename "$pf" .conf)' → gateway '$new'"
        fi
      done
      shopt -u nullglob
      ok "Gateway '$old' renamed → '$new'"
      lock_release
      ;;

    set-token)
      require_interactive
      local name="${1:-}"
      [[ -n "$name" ]] || die "Usage: claude-proxy gateway set-token <n>"
      validate_name "$name" "gateway name"
      require_gateway "$name"
      local gsvc; gsvc="$(read_conf "$GATEWAYS_DIR/${name}.conf" GATEWAY_KEYCHAIN_SERVICE)"
      gsvc="${gsvc:-$(gateway_keychain_service "$name")}"
      echo "Enter new token for gateway '$name':"
      security add-generic-password -a "$name" -s "$gsvc" -U -w
      ok "Token updated in Keychain"
      ;;
```
